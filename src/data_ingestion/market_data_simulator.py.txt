"""
Simple market data simulator for testing
Generates realistic stock price movements
"""

import random
import time
from datetime import datetime
from dataclasses import dataclass, asdict
import json


@dataclass
class MarketTick:
    """Single market data tick"""
    symbol: str
    timestamp: str
    price: float
    volume: int
    bid: float
    ask: float


class MarketDataSimulator:
    """Simulates real-time market data"""
    
    def __init__(self):
        # Popular stocks for portfolio
        self.symbols = ['AAPL', 'GOOGL', 'MSFT', 'AMZN', 'JPM']
        # Starting prices (approximately real as of 2025)
        self.prices = {
            'AAPL': 180.0,
            'GOOGL': 140.0,
            'MSFT': 380.0,
            'AMZN': 170.0,
            'JPM': 150.0
        }
    
    def generate_tick(self, symbol: str) -> MarketTick:
        """Generate one market tick"""
        # Get current price
        current_price = self.prices[symbol]
        
        # Random price movement (-1% to +1%)
        price_change = current_price * random.uniform(-0.01, 0.01)
        new_price = current_price + price_change
        
        # Update stored price
        self.prices[symbol] = new_price
        
        # Generate bid/ask spread (0.01% typically)
        spread = new_price * 0.0001
        bid = new_price - spread
        ask = new_price + spread
        
        # Random volume (1000-10000 shares)
        volume = random.randint(1000, 10000)
        
        return MarketTick(
            symbol=symbol,
            timestamp=datetime.now().isoformat(),
            price=round(new_price, 2),
            volume=volume,
            bid=round(bid, 2),
            ask=round(ask, 2)
        )
    
    def stream_data(self, duration_seconds: int = 60):
        """Stream market data for specified duration"""
        print(f"ðŸš€ Starting market data stream for {duration_seconds} seconds")
        print(f"ðŸ“Š Symbols: {', '.join(self.symbols)}")
        print("-" * 80)
        
        start_time = time.time()
        tick_count = 0
        
        while time.time() - start_time < duration_seconds:
            # Pick random symbol
            symbol = random.choice(self.symbols)
            
            # Generate tick
            tick = self.generate_tick(symbol)
            
            # Print tick
            print(f"{tick.symbol:6} | ${tick.price:7.2f} | "
                  f"Bid: ${tick.bid:7.2f} | Ask: ${tick.ask:7.2f} | "
                  f"Vol: {tick.volume:5} | {tick.timestamp}")
            
            tick_count += 1
            
            # Wait a bit (simulates real-time data)
            time.sleep(0.1)  # 10 ticks per second
        
        elapsed = time.time() - start_time
        tps = tick_count / elapsed
        
        print("-" * 80)
        print(f"âœ… Stream complete!")
        print(f"ðŸ“ˆ Total ticks: {tick_count}")
        print(f"âš¡ Ticks per second: {tps:.1f}")


if __name__ == "__main__":
    # Run the simulator
    simulator = MarketDataSimulator()
    simulator.stream_data(duration_seconds=30)
